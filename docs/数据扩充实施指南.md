### **步骤4: 合并数据池**

当生成3-5批后，合并为完整数据池：

```python
# 使用Python交互模式
import json
from pathlib import Path

def merge_batches(prefix, output_file, batch_count=5):
    """合并批次数据"""
    all_data = []
    
    for i in range(1, batch_count + 1):
        file = Path(f"data/{prefix}-batch{i}.json")
        if file.exists():
            with open(file, 'r', encoding='utf-8') as f:
                data = json.load(f)
                all_data.extend(data)
            print(f"✓ 加载批次{i}: {len(data)} 条")
    
    # 保存合并结果
    with open(f"data/{output_file}", 'w', encoding='utf-8') as f:
        json.dump(all_data, f, ensure_ascii=False, indent=2)
    
    print(f"\n✅ 合并完成: {len(all_data)} 条数据")
    print(f"💾 保存至: {output_file}")

# 执行合并
merge_batches("pool-questions", "pool-questions-500.json", 5)
merge_batches("pool-keywords", "pool-keywords-500.json", 5)
merge_batches("pool-data", "pool-data-500.json", 5)
```
